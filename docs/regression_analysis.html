Z-scoring

z.fun <- function(df,variables) { 
  for (variable in variables) {
  df[paste0(variable,'.z')] <- NULL
  df[paste0(variable,'.z')] <- as.numeric(scale(df[variable]))
  }
  return (df)
}

varbs.for.z = c(
  'n_segments',
  "n_syllables",
  "activation_context",
